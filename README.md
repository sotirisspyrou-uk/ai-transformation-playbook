from ai_transformation import FoundationBuilder

### ROI Measurement Methodology
```python
# Comprehensive ROI calculation
from measurement_optimization import ROICalculator

roi_calculator = ROICalculator()

# Calculate direct financial benefits
direct_benefits = roi_calculator.calculate_direct_benefits([
    {'source': 'process_automation', 'annual_savings': 1200000},
    {'source': 'error_reduction', 'annual_savings': 450000},
    {'source': 'productivity_improvement', 'annual_value': 850000},
    {'source': 'new_revenue_opportunities', 'annual_value': 2100000}
])

# Account for implementation costs
implementation_costs = roi_calculator.calculate_total_costs([
    {'category': 'technology', 'amount': 800000},
    {'category': 'personnel', 'amount': 1200000},
    {'category': 'training', 'amount': 300000},
    {'category': 'consulting', 'amount': 400000}
])

# Calculate 3-year ROI
roi_analysis = roi_calculator.calculate_roi(
    benefits=direct_benefits,
    costs=implementation_costs,
    analysis_period_years=3,
    discount_rate=0.08
)

print(f"Total 3-Year ROI: {roi_analysis['total_roi']:.1%}")
print(f"Payback Period: {roi_analysis['payback_months']} months")
print(f"NPV: ${roi_analysis['npv']:,}")
```
